LIST OF FRIEND REQUESTS)
SELECT CONCAT(accounts.firstName,' ', accounts.lastName) AS 'FRIENDS', dateTime
FROM socials
INNER JOIN accounts ON socials.user2_id = accounts.id
WHERE socials.user1_id = 1

LIST OF BLOCKED FRIENDS
SELECT CONCAT(accounts.firstName,' ', accounts.lastName) AS 'BLOCKED', dateTime FROM blocked_friends INNER JOIN accounts ON blocked_friends.blockedUser_id = accounts.id WHERE blocked_friends.user_id = 1;
